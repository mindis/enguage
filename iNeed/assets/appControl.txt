# how to switch between preview and direct mode
     On " direct",          perform "preferences set oBMode direct".
Then on " direct",          reply   "ok, direct mode is now on".
     On "preview",          perform "preferences set oBMode preview".
Then on "preview",          reply   "ok, preview mode is now on".
     On "direct on",        think   "direct".
     On "direct off",       think   "preview".
     On "preview mode",     think   "preview".
     On "direct mode",      think   "direct".
     On "direct mode on",   think   "direct".
     On "direct mode off",  think   "preview".
     On "preview off",      think   "direct".
     On "preview on",       think   "preview".
     On "preview mode off", think   "direct".
     On "preview mode on",  think   "preview".

On        "repeat", think "say again".
On "please repeat", think "say again".
On "say it again",  think "say again".

On "no", think "undo".

     On "bigger text.", perform "numeric increase device textSize 4".

     On "larger text.", think "bigger text".

     On "smaller text.", perform "numeric decrease device textSize 4".
Then on "smaller text.", reply "ok, text should now be smaller".
